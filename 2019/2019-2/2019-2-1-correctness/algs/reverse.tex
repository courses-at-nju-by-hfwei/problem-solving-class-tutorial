\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{reverse}{$S$}
      \State \uncover<2->{\teal{$P: \top$}}
      \State $X \gets S$  \Comment{keep $S$ intact}
      \State $Y \gets \Lambda$  \Comment{Y is to store the result}
      
      \hStatex
      \State \uncover<3->{\red{$I: S = \texttt{reverse}(Y) \cdot X$}}
      \While{$X \neq \Lambda$}
        \State $Y \gets \Call{head}{X} \cdot Y$
        \State $X \gets \Call{tail}{X}$
      \EndWhile
      \State \uncover<2->{\teal{$Q: Y = \texttt{reverse}(S)$}}
      
      \hStatex
      \State \Return Y
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
