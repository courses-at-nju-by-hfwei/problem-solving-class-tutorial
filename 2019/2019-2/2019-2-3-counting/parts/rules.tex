\begin{frame}{}
  \begin{theorem}[Sum Principle]
    \[
      S \cap T = \emptyset \implies |S \cup T| = |S| + |T|
    \]
  \end{theorem}

  \pause
  \vspace{0.60cm}
  \begin{theorem}[Product Principle]
    \[
      |S \times T| = |S| \times |T|
    \]
  \end{theorem}

  \pause
  \vspace{0.60cm}
  \begin{center}
    Holds for \red{finite} sets $S$ and $T$.
  \end{center}
\end{frame}
