\begin{frame}{}
  \[
    x_1 + x_2 + \cdots + x_k = n
  \]
  \[
    x_1 \ge x_2 \ge \cdots \ge x_k \ge 1
  \]

  \begin{theorem}[Recurrence for $\partition{n}{k}$]
    \[
      \partition{n}{k} = \partition{n-1}{k-1} + \partition{n}{n-k}
    \]
  \end{theorem}

  \pause
  \begin{columns}
    \column{0.40\textwidth}
    \[
      \red{\text{\textsc{Case I}}: x_k = 1}
    \]

    \uncover<3->{
      \[
        \blue{\boxed{\partition{n-1}{k-1}}}
      \]
    }
    \column{0.60\textwidth}
      \[
        \red{\text{\textsc{Case II}}: x_k > 1}
      \]
      \uncover<4->{
        \[
          x_1 - 1 \ge x_2 - 1 \ge \cdots \ge x_k - 1 \ge 1
        \]
      }
      \uncover<5->{
        \[
          \blue{\boxed{\partition{n}{n-k}}}
        \]
      }
  \end{columns}
\end{frame}
